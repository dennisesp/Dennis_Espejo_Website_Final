)

# Function to Update Chart
def update_chart(field):
    field_data = aggregated_data.loc[field].sort_values(ascending=True)  # Sort for better visualization
    
    fig = px.bar(
        y=field_data.index,
        x=field_data.values,
        orientation="h",

    )
    
    fig.show()
